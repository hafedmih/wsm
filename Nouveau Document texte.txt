CREATE TABLE `comments` (
  `id` int(11) UNSIGNED NOT NULL AUTO_INCREMENT,
  
  -- Polymorphism: Links to Project ID or Meeting ID
  `resource_id` int(11) NOT NULL COMMENT 'The ID of the project or meeting',
  `resource_type` varchar(20) NOT NULL DEFAULT 'project' COMMENT 'Values: project, meeting',
  
  -- Threading: 0 = Main Comment, >0 = Reply to that ID
  `parent_id` int(11) DEFAULT 0,
  
  -- User Info
  `user_id` int(11) NOT NULL,
  `role` varchar(50) DEFAULT NULL COMMENT 'Stored for fast styling: admin, donor, ministry',
  
  -- Content
  `message` longtext NOT NULL,
  `attachment` varchar(255) DEFAULT NULL COMMENT 'Optional: for future file uploads',
  
  -- System
  `status` int(1) DEFAULT 1 COMMENT '1=Active, 0=Hidden/Deleted',
  `created_at` int(11) DEFAULT NULL,
  `updated_at` int(11) DEFAULT NULL,
  
  PRIMARY KEY (`id`),
  
  -- Indexes for Performance (Crucial for speed)
  KEY `idx_resource` (`resource_id`, `resource_type`), 
  KEY `idx_parent` (`parent_id`)
  
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

  `agreement_date` varchar(20) DEFAULT NULL,
  `created_at` int(11) DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
